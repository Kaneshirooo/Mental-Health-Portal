    users {
        bigint user_id PK
        string email
        string password
        string full_name
        string roll_number
        enum user_type
        date date_of_birth
        enum gender
        string contact_number
        string department
        integer semester
        timestamp created_at
        timestamp updated_at
    }

    otp_tokens {
        integer id PK
        integer user_id FK
        string otp_code
        datetime expires_at
        boolean used
        timestamp created_at
    }

    appointments {
        bigint appointment_id PK
        bigint student_id FK
        bigint counselor_id FK
        datetime scheduled_at
        integer duration_min
        enum status
        text reason
        text counselor_message
        boolean is_priority
        timestamp created_at
        timestamp updated_at
    }

    ai_preassessments {
        bigint pre_id PK
        bigint student_id FK
        mediumText conversation_transcript
        text form_answers
        text ai_report
        timestamp created_at
    }

    counselor_availability {
        bigint availability_id PK
        bigint counselor_id FK
        tinyInteger day_of_week
        time start_time
        time end_time
        boolean is_active
        timestamp created_at
        timestamp updated_at
    }

    assessment_questions {
        bigint question_id PK
        string category
        text question_text
        integer question_number
        timestamp created_at
    }

    student_responses {
        bigint response_id PK
        bigint user_id FK
        bigint question_id FK
        integer response_value
        timestamp assessment_date
    }

    assessment_scores {
        bigint score_id PK
        bigint user_id FK
        integer depression_score
        integer anxiety_score
        integer stress_score
        integer overall_score
        enum risk_level
        timestamp assessment_date
        timestamp report_generated_at
        text counselor_notes
    }

    mood_logs {
        bigint log_id PK
        bigint student_id FK
        tinyInteger mood_score
        string mood_emoji
        text note
        timestamp logged_at
    }

    anonymous_notes {
        bigint note_id PK
        bigint student_id FK
        text message
        text reply
        timestamp replied_at
        bigint counselor_id FK
        enum status
        timestamp created_at
    }

    anonymous_note_messages {
        bigint message_id PK
        bigint note_id FK
        enum sender_type
        text message_text
        timestamp created_at
    }

    counselor_notes {
        bigint note_id PK
        bigint counselor_id FK
        bigint student_id FK
        text note_text
        string recommendation
        date follow_up_date
        timestamp created_at
        timestamp updated_at
    }

    chat_history {
        bigint chat_id PK
        bigint sender_id FK
        bigint receiver_id FK
        text message
        boolean is_read
        timestamp created_at
    }

    notifications {
        uuid id PK
        string type
        string notifiable_type
        bigint notifiable_id
        text data
        timestamp read_at
        timestamp created_at
        timestamp updated_at
    }

    session_logs {
        bigint log_id PK
        bigint user_id FK
        string action
        string ip_address
        text user_agent
        timestamp created_at
    }
